			public KeyDownBehavior(){
				super("onkeydown");
			}
			@Override
			protected void onEvent(AjaxRequestTarget arg0) {
				System.out.println("event on");
				try{
					Thread.sleep(2000);
				}catch(Exception ex){};
				System.out.println("event off");
			}

			@Override
			public String getAjaxIndicatorMarkupId() {
				return "nick_indicator";
			}
